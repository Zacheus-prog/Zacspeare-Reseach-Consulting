Research Consultancy Web const { useState } = React; 
// Service data with rates
const services = [
  { id: 1, name: "Questionnaire Designing", rate: 500, description: "Tailored questionnaire creation for precise research outcomes." },
  { id: 2, name: "Questionnaire Testing", rate: 300, description: "Validation and testing to ensure questionnaire reliability." },
  { id: 3, name: "Data Collection", rate: 1000, description: "Multi-channel data collection for robust datasets." },
  { id: 4, name: "Data Quality Control & Assurance", rate: 400, description: "Rigorous checks to maintain data accuracy and integrity." },
  { id: 5, name: "Data Cleaning", rate: 350, description: "Data preparation for error-free analysis." },
  { id: 6, name: "Data Analysis", rate: 800, description: "Advanced statistical analysis with actionable insights." },
  { id: 7, name: "Report Presentation", rate: 600, description: "Professional reports with clear, visual presentations." },
  { id: 8, name: "Overall Research Consultancy", rate: 2000, description: "Comprehensive research support from start to finish." },
];

// Service Card Component
function ServiceCard({ service, addToOrder }) {
  return (
    <div className="bg-white p-6 rounded-lg shadow-md hover:shadow-xl transition-all">
      <h3 className="text-xl font-semibold text-gray-900">{service.name}</h3>
      <p className="text-gray-700 mt-2">{service.description}</p>
      <p className="text-lg font-bold text-teal-600 mt-2">${service.rate}</p>
      <button
        onClick={() => addToOrder(service)}
        className="mt-4 bg-teal-500 text-white px-4 py-2 rounded-lg hover:bg-teal-600 transition"
      >
        Add to Order
      </button>
    </div>
  );
}

// Order Summary Component
function OrderSummary({ order, removeFromOrder, submitOrder }) {
  const total = order.reduce((sum, item) => sum + item.rate, 0);

  return (
    <div className="bg-gray-50 p-6 rounded-lg shadow-md">
      <h2 className="text-2xl font-bold text-gray-900 mb-4">Order Summary</h2>
      {order.length === 0 ? (
        <p className="text-gray-700">No services selected.</p>
      ) : (
        <div>
          {order.map((item, index) => (
            <div key={index} className="flex justify-between items-center mb-2">
              <span className="text-gray-800">{item.name}</span>
              <div className="flex items-center">
                <span className="mr-4 text-teal-600">${item.rate}</span>
                <button
                  onClick={() => removeFromOrder(index)}
                  className="text-red-600 hover:text-red-800"
                >
                  Remove
                </button>
              </div>
            </div>
          ))}
          <div className="mt-4 border-t pt-2">
            <p className="text-lg font-bold text-gray-900">Total: ${total}</p>
          </div>
          <button
            onClick={submitOrder}
            className="mt-4 bg-teal-600 text-white px-4 py-2 rounded-lg hover:bg-teal-700 transition"
          >
            Submit Order
          </button>
        </div>
      )}
    </div>
  );
}

// Main App Component
function App() {
  const [order, setOrder] = useState([]);

  const addToOrder = (service) => {
    setOrder([...order, service]);
  };

  const removeFromOrder = (index) => {
    setOrder(order.filter((_, i) => i !== index));
  };

  const submitOrder = () => {
    if (order.length === 0) {
      alert("Please select at least one service.");
      return;
    }
    alert("Order submitted successfully! Our team will contact you soon.");
    setOrder([]);
  };

  return (
    <div className="min-h-screen bg-gray-100 p-8">
      <h1 className="text-4xl font-bold text-center text-gray-900 mb-8">
        Research Consultancy Web
      </h1>
      <div className="max-w-6xl mx-auto grid grid-cols-1 lg:grid-cols-2 gap-6">
        <div>
          <h2 className="text-2xl font-bold text-gray-900 mb-4">Our Services</h2>
          <div className="grid grid-cols-1 gap-4">
            {services.map((service) => (
              <ServiceCard key={service.id} service={service} addToOrder={addToOrder} />
            ))}
          </div>
        </div>
        <div>
          <OrderSummary order={order} removeFromOrder={removeFromOrder} submitOrder={submitOrder} />
        </div>
      </div>
    </div>
  );
}

// Render the app
ReactDOM.render(<App />, document.getElementById('root'));
